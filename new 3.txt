let currentMode = 'gregorian';

function switchCalendar(mode) {
    currentMode = mode;
    document.getElementById('btn-gregorian').classList.toggle('active', mode === 'gregorian');
    document.getElementById('btn-hijri').classList.toggle('active', mode === 'hijri');
    
    if (mode === 'gregorian') {
        document.getElementById('birth-date').classList.remove('hidden');
        document.getElementById('hijri-inputs').classList.add('hidden');
    } else {
        document.getElementById('birth-date').classList.add('hidden');
        document.getElementById('hijri-inputs').classList.remove('hidden');
    }
}

// دالة تقريبية متطورة لتحويل التاريخ الهجري إلى ميلادي لأغراض الحساب الدقيق
function hijriToGregorian(hDay, hMonth, hYear) {
    // معادلة تحويل تقريبية معتمدة برمجياً لضبط الفوارق
    let mYear = Math.floor(hYear * 0.970229) + 621.5643;
    let mMonth = hMonth; 
    let mDay = hDay;
    
    // ضبط الأيام بشكل تقريبي جداً للبدء بالحساب عبر نظام الـ Date الافتراضي
    let jd = Math.floor((11 * hYear + 3) / 30) + 354 * hYear + 30 * hMonth - Math.floor((hMonth - 1) / 2) + hDay + 1948440 - 385;
    let l = jd + 68569;
    let n = Math.floor((4 * l) / 146097);
    l = l - Math.floor((146097 * n + 3) / 4);
    let i = Math.floor((4000 * (l + 1)) / 1461001);
    l = l - Math.floor((1461 * i) / 4) + 31;
    let j = Math.floor((80 * l) / 2447);
    let d = l - Math.floor((2447 * j) / 80);
    l = Math.floor(j / 11);
    let m = j + 2 - 12 * l;
    let y = 100 * (n - 49) + i + l;
    
    return new Date(y, m - 1, d);
}

function calculateAge() {
    let birthDate;
    const today = new Date();

    if (currentMode === 'gregorian') {
        const dateInput = document.getElementById('birth-date').value;
        if (!dateInput) {
            alert('الرجاء اختيار تاريخ الميلاد الميلادي');
            return;
        }
        birthDate = new Date(dateInput);
    } else {
        const hDay = parseInt(document.getElementById('hijri-day').value);
        const hMonth = parseInt(document.getElementById('hijri-month').value);
        const hYear = parseInt(document.getElementById('hijri-year').value);

        if (!hDay || !hMonth || !hYear) {
            alert('الرجاء إدخال التاريخ الهجري كاملاً بشكل صحيح');
            return;
        }
        birthDate = hijriToGregorian(hDay, hMonth, hYear);
    }

    if (birthDate > today) {
        alert('تاريخ الميلاد لا يمكن أن يكون في المستقبل!');
        return;
    }

    // حساب الفروق بدقة
    let years = today.getFullYear() - birthDate.getFullYear();
    let months = today.getMonth() - birthDate.getMonth();
    let days = today.getDate() - birthDate.getDate();

    if (days < 0) {
        months--;
        // جلب عدد الأيام في الشهر السابق
        const previousMonth = new Date(today.getFullYear(), today.getMonth(), 0).getDate();
        days += previousMonth;
    }

    if (months < 0) {
        years--;
        months += 12;
    }

    // عرض النتائج في الواجهة
    document.getElementById('res-years').innerText = years;
    document.getElementById('res-months').innerText = months;
    document.getElementById('res-days').innerText = days;

    // حساب الأيام المتبقية ليوم الميلاد القادم (ميلادياً كمؤشر)
    let nextBday = new Date(today.getFullYear(), birthDate.getMonth(), birthDate.getDate());
    if (today > nextBday) {
        nextBday.setFullYear(today.getFullYear() + 1);
    }
    let diffTime = Math.abs(nextBday - today);
    let diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
    
    if(diffDays === 365 || diffDays === 0) {
        document.getElementById('next-birthday').innerText = "🎉 عيد ميلاد سعيد! اليوم هو يوم ميلادك.";
    } else {
        document.getElementById('next-birthday').innerText = `⏳ المتبقي على يوم ميلادك القادم: ${diffDays} يوماً.`;
    }

    document.getElementById('result').classList.remove('hidden');
}
